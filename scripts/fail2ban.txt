[sshd]
enabled = true
mode = aggressive
port = ssh
logpath = %(sshd_log)s
maxretry = 4
findtime = 15m
bantime = 12h


# deploy config
sudo install -d /etc/fail2ban/jail.d
sudo install -m 0644 config/fail2ban/jail.d/sshd.local /etc/fail2ban/jail.d/sshd.local
sudo systemctl restart fail2ban
sudo fail2ban-client status sshd
